@bot.message_handler(commands=['start'])
def handle_start(message):
    user_id = message.chat.id

    markup = types.ReplyKeyboardMarkup(row_width=2)
    btn1 = types.KeyboardButton('crazy')
    btn2 = types.KeyboardButton('неформальное')
    btn3 = types.KeyboardButton('формальное')
    markup.add(btn1, btn2, btn3)
    bot.send_message(user_id, "Привет! Выбери тип общения:", reply_markup=markup)
